数据源管理模块的数据库结构

kname char(100) not null,/*数据源名称*/ ntype tinyint not null default 0,/*数据源类型 MySQL or SQLite*/ dbhost char(50) null,/*数据库地址*/ dbname char(50) null,/*数据库名称*/ dbuser char(50) null,/*数据库帐号*/ dbpass char(50) null,/*数据库密码*/ dbfile char(100) null,/*数据库文件 SQLite时候*/ dbcharset char(20) null,/*数据库编码*/ norder int not null default 0/*排序*/
